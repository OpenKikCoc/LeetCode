### [101061. Concatenate Array With Reverse](https://leetcode.cn/problems/concatenate-array-with-reverse/)

```c++
class Solution {
public:
    vector<int> concatWithReverse(vector<int>& nums) {
        int n = nums.size();
        for (int i = n - 1; i >= 0; -- i )
            nums.push_back(nums[i]);
        return nums;
    }
};
```

### [101055. Count Valid Word Occurrences](https://leetcode.cn/problems/count-valid-word-occurrences/)

```c++
class Solution {
public:
    bool check(string & s, int x, int y) {
        bool f1 = isalpha(s[x]), f2 = isalpha(s[y]);
        if (f1 && f2)
            return true;
        if (f1 != f2)
            return true;
        return false;
    }
    
    vector<int> countWordOccurrences(vector<string>& chunks, vector<string>& queries) {
        string s;
        for (auto & c : chunks)
            s += c;

        int n = s.size();
        unordered_map<string, int> hash;
        for (int i = 0; i < n; ++ i ) {
            if (s[i] == ' ' || s[i] == '-')
                continue;
            // 一定是字母开头
            int j = i;
            while (j < n && s[j] != ' ' && check(s, j, j + 1))
                j ++ ;
            string t = s.substr(i, j - i);
            // cout << " t = " << t << endl;
            hash[t] ++ ;
            i = j;
        }
        
        vector<int> res;
        for (auto & q: queries)
            res.push_back(hash[q]);
        return res;
    }
};
```

### [101063. Minimize Array Sum Using Divisible Replacements](https://leetcode.cn/problems/minimize-array-sum-using-divisible-replacements/)

```c++
class Solution {
public:
    using LL = long long;
    // 本质为将一个数的倍数替换为这个数本身 显然是用越小的替换越好
    // 考虑顺序无关 且值域有限
    const static int N = 1e5 + 10;

    int c[N];
    
    long long minArraySum(vector<int>& nums) {
        memset(c, 0, sizeof c);
        for (auto x : nums)
            c[x] ++ ;
        
        for (int i = 1; i < N; ++ i )
            if (c[i])
                for (int j = i + i; j < N; j += i )
                    c[i] += c[j], c[j] = 0;

        LL res = 0;
        for (int i = 1; i < N; ++ i )
            res += (LL)i * c[i];
        return res;
    }
};
```

### [100757. Minimum Cost to Buy Apples II](https://leetcode.cn/problems/minimum-cost-to-buy-apples-ii/)

```c++
const static int N = 1010, M = 2010; // ATTENTION M
using LL = long long; // ATTENTION price
using PII = pair<LL, int>;

int h1[N], h2[N], e[M << 2], ne[M << 2], idx;
LL w[M << 2];
void add(int h[], int a, int b, LL c) {
    e[idx] = b, w[idx] = c, ne[idx] = h[a], h[a] = idx ++ ;
}

LL d1[N][N], d2[N][N];
bool st[N];
void init() {
    memset(h1, -1, sizeof h1), memset(h2, -1, sizeof h2);
    idx = 0;
    memset(d1, 0x3f, sizeof d1), memset(d2, 0x3f, sizeof d2);
}

class Solution {
public:
    void dijkstra(int start, LL base, LL d[], int h[]) {
        d[start] = base;
        memset(st, 0, sizeof st);

        priority_queue<PII, vector<PII>, greater<PII>> heap;
        heap.push({(LL)base, start});
        while (heap.size()) {
            auto [base, u] = heap.top();
            heap.pop();

            if (st[u])
                 continue;
            st[u] = true;
            for (int i = h[u]; ~i; i = ne[i]) {
                int j = e[i];
                if (d[j] > d[u] + w[i]) {
                    d[j] = d[u] + w[i];
                    heap.push({d[j], j});
                }
            }
        }
    }
    
    vector<int> minCost(int n, vector<int>& prices, vector<vector<int>>& roads) {
        init();
        for (auto & r : roads) {
            int a = r[0], b = r[1], c1 = r[2], c2 = r[3];
            add(h1, a, b, c1), add(h1, b, a, c1);
            add(h2, a, b, (LL)c1 * c2), add(h2, b, a, (LL)c1 * c2);
        }


        // 稀疏图
        // 反向思考 在某个位置买苹果 去到其它位置的最小开销

        for (int i = 0; i < n; ++ i ) {
            // 1. 从 i 出发不带苹果 到不同位置的最小值
            dijkstra(i, 0, d1[i], h1);
            // 2. 从 i 出发带苹果 到不同位置的最小值
            dijkstra(i, prices[i], d2[i], h2);
        }

        vector<int> res;
        for (int i = 0; i < n; ++ i ) {
            LL x = prices[i];
            for (int j = 0; j < n; ++ j )
                x = min(x, d1[i][j] + d2[j][i]);
            res.push_back(x);
        }
        return res;
    }
};
```
