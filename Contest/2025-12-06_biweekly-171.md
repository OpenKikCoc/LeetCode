### [3765. Complete Prime Number](https://leetcode.cn/problems/complete-prime-number/)

细节比较多 非空串才有判断的必要 WA2

```c++
class Solution {
public:
    bool check(int x) {
        if (x < 2)
            return false;
        for (int i = 2; i <= x / i; ++ i )
            if (x % i == 0)
                return false;
        return true;
    }
    
    bool completePrime(int num) {
        string s = to_string(num);
        int n = s.size();
        
        for (int i = 0; i <= n; ++ i ) {
            int a = 0, b = 0;
            for (int j = 0; j < i; ++ j )
                a = a * 10 + s[j] - '0';
            for (int j = i; j < n; ++ j )
                b = b * 10 + s[j] - '0';
            if (i && !check(a)) // ATTENTION
                return false;
            if (i < n && !check(b))
                return false;
        }
        return true;
    }
};
```

### [3766. Minimum Operations to Make Binary Palindrome](https://leetcode.cn/problems/minimum-operations-to-make-binary-palindrome/)

想的太复杂了 实际上这个数据范围在预处理时直接暴力枚举就 ok

```c++
bool check(int x) {
    vector<int> xs;
    while (x)
        xs.push_back(x % 2), x /= 2;
    int n = xs.size();
    for (int i = 0, j = n - 1; i < j; ++ i , -- j )
        if (xs[i] != xs[j])
            return false;
    return true;
}
vector<int> xs;
void init() {
    if (xs.size())
        return;
    int l = 1, r = 1e4 + 10;
    for (int i = l; i <= r; ++ i )
        if (check(i))
            xs.push_back(i);
}

class Solution {
public:
    int get(int x) {
        auto it = lower_bound(xs.begin(), xs.end(), x);
        int ret = (*it) - x;
        if (it != xs.begin()) {
            it -- ;
            ret = min(ret, x - (*it));
        }
        return ret;
    }
    vector<int> minOperations(vector<int>& nums) {
        init();
        vector<int> res;
        for (auto x : nums)
            res.push_back(get(x));
        return res;
    }
};
```

### [3767. Maximize Points After Choosing K Tasks](https://leetcode.cn/problems/maximize-points-after-choosing-k-tasks/)

显然贪心 选差值最大的 k 个

细节思考: 如果差值相同 选不同的下标是否有影响? => 没有 本质总贡献是一致的 所以不需要考虑

```c++
class Solution {
public:
    using LL = long long;
    using PII = pair<int, int>;
    
    long long maxPoints(vector<int>& technique1, vector<int>& technique2, int k) {
        int n = technique1.size();

        LL res = 0;
        // 显然选择1更有优势的完成
        priority_queue<PII, vector<PII>, greater<PII>> heap; // 小顶堆
        for (int i = 0; i < n; ++ i ) {
            int x = technique1[i] - technique2[i];
            heap.push({x, i}); res += technique1[i];
            if (heap.size() > k) {
                auto [y, j] = heap.top();
                heap.pop(); res += max(technique1[j], technique2[j]) - technique1[j];
            }
        }
        return res;
    }
};
```

### [3768. Minimum Inversion Count in Subarrays of Fixed Length](https://leetcode.cn/problems/minimum-inversion-count-in-subarrays-of-fixed-length/)

```c++
const static int N = 1e5 + 10;
int tr[N];
void init() {
    memset(tr, 0, sizeof tr);
}
int lowbit(int x) {
    return x & -x;
}
void add(int x, int c) {
    for (int i = x; i < N; i += lowbit(i))
        tr[i] += c;
}
int sum(int x) {
    if (x <= 0) // 特判下
        return 0;
    int ret = 0;
    for (int i = x; i; i -= lowbit(i))
        ret += tr[i];
    return ret;
}

class Solution {
public:
    using LL = long long;
    
    vector<int> xs;
    int get(int x) {
        return lower_bound(xs.begin(), xs.end(), x) - xs.begin() + 1;
    }

    long long minInversionCount(vector<int>& nums, int k) {
        if (k == 1)
            return 0;

        // 离散化
        {
            for (auto x : nums)
                xs.push_back(x);
            sort(xs.begin(), xs.end());
            xs.erase(unique(xs.begin(), xs.end()), xs.end());
        }

        LL res = 1e12;
        {
            init();
            int n = nums.size();
            LL tot = 0;
            for (int i = 0, j = 0; i < n; ++ i ) {
                int x = get(nums[i]);
                // 先读再写
                if (i - j + 1 > k) {
                    int y = get(nums[j]);
                    j ++ ;
                    tot -= sum(y - 1);
                    add(y, -1);
                }
                tot += sum(N - 1) - sum(x);
                add(x, 1);
                if (i - j + 1 == k) {
                    // cout << " i = " << i << " j = " << j << " tot = " << tot << endl;
                    res = min(res, tot);
                }
            }
        }
        return res;
    }
};
```
