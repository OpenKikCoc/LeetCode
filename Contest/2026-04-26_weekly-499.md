### [?. Valid Elements in an Array](https://leetcode.cn/problems/valid-elements-in-an-array/)

```c++
class Solution {
public:
    vector<int> findValidElements(vector<int>& nums) {
        vector<int> res;
        int n = nums.size();
        for (int i = 0; i < n; ++ i ) {
            int l = 0, r = 0;
            for (int j = 0; j < i; ++ j )
                l = max(l, nums[j]);
            for (int j = i + 1; j < n; ++ j )
                r = max(r, nums[j]);
            if (nums[i] > l || nums[i] > r)
                res.push_back(nums[i]);
        }
        return res;
    }
};
```

### [?. Sort Vowels by Frequency](https://leetcode.cn/problems/sort-vowels-by-frequency/)

```c++
class Solution {
public:
    bool check(char c) {
        return c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u';
    }
    
    string sortVowels(string s) {
        int n = s.size();
        unordered_map<char, int> hash, idx;
        for (int i = 0; i < n; ++ i ) {
            char c = s[i];
            if (check(c)) {
                hash[c] ++ ;
                if (!idx.count(c))
                    idx[c] = i;
            }            
        }

        vector<tuple<int, int, char>> t;
        for (auto [c, cnt] : hash)
            t.push_back({-cnt, idx[c], c});
        sort(t.begin(), t.end());
        
        for (int i = 0; i < n; ++ i ) {
            char c = s[i];
            if (!check(c))
                continue;
            for (auto & [neg_cnt, _, c] : t)
                if (neg_cnt) {
                    s[i] = c;
                    neg_cnt ++ ;
                    break;
                }
        }
        return s;
    }
};
```

### [?. Minimum Operations to Make Array Non Decreasing](https://leetcode.cn/problems/minimum-operations-to-make-array-non-decreasing/)

```c++
class Solution {
public:
    using LL = long long;

    long long minOperations(vector<int>& nums) {
        int n = nums.size();
        LL res = 0;
        for (int i = n - 2; i >= 0; -- i )
             if (nums[i] > nums[i + 1])
                 res += nums[i] - nums[i + 1];
        return res;
    }
};
```

### [?. Maximum Sum of Alternating Subsequence With Distance at Least K](https://leetcode.cn/problems/maximum-sum-of-alternating-subsequence-with-distance-at-least-k/)

```c++
using LL = long long;

const static int N = 1e5 + 10;
LL w1[N], w2[N];
LL tr1[N], tr2[N];

int lowbit(int x) {
    return x & -x;
}
void modify(LL w[], LL tr[], int x, LL y) {
    w[x] = y;
    for (int i = x; i < N; i += lowbit(i))
        tr[i] = max(tr[i], y);
}
LL query(LL w[], LL tr[], int l, int r) {
    LL res = 0;
    for (; l <= r; ) {
        res = max(res, w[r]);
        for ( -- r ; r >= l + lowbit(r); r -= lowbit(r))
            res = max(res, tr[r]);
    }
    return res;
}

LL f[N][2];

void init() {
    memset(w1, 0, sizeof w1), memset(w2, 0, sizeof w2);
    memset(tr1, 0, sizeof tr1), memset(tr2, 0, sizeof tr2);
    memset(f, 0, sizeof f);
}

class Solution {
public:
    // 显然是动态规划
    // f[i][0] 表示以第i个数字为结尾 之前是上升的最大和
    // f[i][1] 表示以第i个数字为结尾 之前是下降的最大和
    // 
    // f[i][0] = max(f[j][1]) + nums[i], j <= i-k 且 nums[j] > nums[i] 的最大和
    // 复杂度在于查询最值 显然可以 bit (值域)
    

    long long maxAlternatingSum(vector<int>& nums, int k) {
        init();
        int n = nums.size();
        LL res = 0;
        for (int i = 0; i < n; ++ i ) {
            if (i - k >= 0) {
                int y = i - k;
                int x = nums[y]; // ATTENTION
                if (w1[x] < f[y][0])
                    modify(w1, tr1, x, f[y][0]);
                if (w2[x] < f[y][1])
                    modify(w2, tr2, x, f[y][1]);
            }
            int x = nums[i];
            f[i][0] = query(w2, tr2, 1, x - 1) + x;
            f[i][1] = query(w1, tr1, x + 1, N - 1) + x;
            res = max(res, max(f[i][0], f[i][1]));
            // cout << " i = " << i << " x = " << x << " 0 = " << f[i][0] << " 1 = " << f[i][1] << endl;
        }
        return res;
    }
};
```

```c++
using LL = long long;
const static int N = 1e5 + 10;

template <typename T>
class BIT {
private:
    int n;
    vector<T> w, tr;
    inline int lowbit(int x) {
        return x & -x;
    }
public:
    BIT(int _n, T init_val = 0) : n(_n) {
        w.assign(n + 1, init_val);
        tr.assign(n + 1, init_val);
    }
    void modify(int x, T y) {
        if (x <= 0 || x > n)
            return;
        if (w[x] >= y) // 特殊
            return;
        w[x] = y;
        for (int i = x; i <= n; i += lowbit(i))
            tr[i] = max(tr[i], y);
    }
    T query(int l, int r) {
        if (l > r || l <= 0 || r > n)
            return 0;
        T res = w[r];
        for ( ; l <= r; ) {
            res = max(res, w[r]);
            for ( -- r ; r >= l + lowbit(r); r -= lowbit(r))
                res = max(res, tr[r]);
        }
        return res;
    }
};

LL f[N][2]; // 本次最后是上升/下降

class Solution {
public:
    long long maxAlternatingSum(vector<int>& nums, int k) {
        int n = nums.size();
        BIT<LL> t1(N), t2(N);
        LL res = 0;
        for (int i = 0; i < n; ++ i ) {
            if (i - k >= 0) {
                int x = i - k, y = nums[x];
                t1.modify(y, f[x][0]), t2.modify(y, f[x][1]);
            }
            int y = nums[i];
            f[i][0] = t2.query(1, y - 1) + y;
            f[i][1] = t1.query(y + 1, N - 1) + y;
            res = max(res, max(f[i][0], f[i][1]));
        }
        return res;
    }
};
```
