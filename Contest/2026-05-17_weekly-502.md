### [3931. Check Adjacent Digit Differences](https://leetcode.cn/problems/check-adjacent-digit-differences/)

```c++
class Solution {
public:
    bool isAdjacentDiffAtMostTwo(string s) {
        int n = s.size();
        for (int i = 1; i < n; ++ i )
            if (abs(s[i] - s[i - 1]) > 2)
                return false;
        return true;
    }
};
```

### [3932. Count K-th Roots in a Range](https://leetcode.cn/problems/count-k-th-roots-in-a-range/)

```c++
class Solution {
public:
    bool k_pow(int m, int k, int x) {
        long long ret = 1;
        for (int i = 0; i < k; ++ i ) {
            ret = ret * m;
            if (ret > x)
                return false;
        }
        return true;
    }
    
    // [0, k] 之间 k 次幂的数量
    //   k = 1, return x+1
    //   k > 1, 不超过 1e5
    int get(int x, int k) {
        if (x < 0)
            return 0;
        if (k == 1)
            return x + 1;
        int l = 1, r = sqrt(x) + 2;
        while (l < r) {
            int mid = l + (r - l) / 2;
            if (k_pow(mid, k, x))
                l = mid + 1;
            else
                r = mid;
        }
        return l;
    }
    
    int countKthRoots(int l, int r, int k) {
        return get(r, k) - get(l - 1, k);
    }
};
```

### [3933. Largest Local Values in a Matrix II](https://leetcode.cn/problems/largest-local-values-in-a-matrix-ii/)

```c++
const static int N = 350, M = 8; // ATTENTION

int f[N][N][M];

void build(vector<vector<int>> & g) {
    memset(f, 0, sizeof f); // ATTENTION
    int n = g.size(), m = g[0].size();
    for (int k = 0; k < M; ++ k )
        for (int i = 0; i < n; ++ i )
            for (int j = 0; j < m; ++ j )
                if (k == 0)
                    f[i][j][0] = g[i][j];
                else {
                    f[i][j][k] = max(f[i][j][k - 1], f[i][j + (1 << (k - 1))][k - 1]);
                }
}

int query(int f[][M], int l, int r) {
    int len = r - l + 1;
    // int k = log(len) / log(2);
    int k = 31 - __builtin_clz(len);
    return max(f[l][k], f[r - (1 << k) + 1][k]);
}

class Solution {
public:
    int countLocalMaximums(vector<vector<int>>& matrix) {
        int n = matrix.size(), m = matrix[0].size();
        build(matrix);
        
        int res = 0;
        for (int i = 0; i < n; ++ i )
            for (int j = 0; j < m; ++ j ) {
                int v = matrix[i][j];
                if (v == 0)
                    continue;

                bool flag = true;
                int x1 = max(0, i - v), x2 = min(n - 1, i + v);
                for (int x = x1; x <= x2; ++ x ) {
                    bool edge = (abs(x - i) == v);
                    int l = max(0, j - v + edge), r = min(m - 1, j + v - edge);
                    if (l > r)
                        continue;
                    if (query(f[x], l, r) > v) {
                        flag = false;
                        break;
                    }
                }
                
                res += flag;
            }
        return res;
    }
};
```

### [3934. Smallest Unique Subarray](https://leetcode.cn/problems/smallest-unique-subarray/)

```c++
const static int N = 1e5 + 10;
using ULL = unsigned long long;
const ULL B = 13331;

ULL h[N], p[N];
ULL get(int l, int r) {
    return h[r] - h[l - 1] * p[r - l + 1];
}

class Solution {
public:
    vector<int> ns;
    int n;
    
    void init() {
        p[0] = 1, h[0] = 0;
        for (int i = 1; i <= n; ++ i ) {
            p[i] = p[i - 1] * B;
            h[i] = h[i - 1] * B + ns[i - 1];
        }
    }
    
    bool check(int x) {
        unordered_map<ULL, int> cnt;
        for (int i = 1; i + x - 1 <= n; ++ i )
            cnt[get(i, i + x - 1)] ++ ;
        for (auto & [k, v] : cnt)
            if (v == 1)
                return true;
        return false;
    }
    
    int smallestUniqueSubarray(vector<int>& nums) {
        this->ns = nums; this->n = ns.size();
        init();

        int l = 1, r = n + 1;
        while (l < r) {
            int mid = l + (r - l) / 2;
            if (check(mid) == false)
                l = mid + 1;
            else
                r = mid;
        }
        return l;
    }
};
```
